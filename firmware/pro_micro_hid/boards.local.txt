# Arduino IDE auto-applies overrides from boards.local.txt sitting next to the
# core's boards.txt. We can't easily change boards.txt without affecting other
# projects, so this file is kept here for reference. To take effect, copy it
# into:
#
#   <Arduino>/hardware/arduino/avr/boards.local.txt
#
# (on Windows that's typically:
#   C:\Users\<you>\AppData\Local\Arduino15\packages\arduino\hardware\avr\<ver>\boards.local.txt
# or wherever your Arduino15 packages dir lives.)
#
# Why: we want the OS to see this board as a generic "HID Keyboard" composite
# device — not "Arduino Micro" or "Arduino Leonardo". Anti-cheats and tooling
# that fingerprint by USB metadata are far less likely to single us out for a
# plain HID name.

# Arduino Micro (atmega32u4) — this is the board we're using.
micro.build.usb_product="HID Keyboard"
micro.build.usb_manufacturer="HID"

# Also cover Leonardo since it's the most common drop-in for atmega32u4 sketches.
leonardo.build.usb_product="HID Keyboard"
leonardo.build.usb_manufacturer="HID"

# PID change is what actually moves the OS off "Arduino Micro" — Arduino's
# driver INF matches by VID/PID and overrides whatever USB iProduct we put on
# the wire. Pick a PID that's outside Arduino's published range so the INF
# doesn't claim it; the device then shows up as a plain composite device with
# our chosen iProduct ("HID Keyboard"). VID stays 0x2341 so we're not spoofing
# another vendor.
#
# Note: the bootloader uses a different PID (0x0037 for Micro, 0x0036 for
# Leonardo). Don't change those — the IDE uses the bootloader PID to find the
# board during upload. Only the sketch-mode PID needs to move.
micro.build.pid=0xC0FE
leonardo.build.pid=0xC0FF
